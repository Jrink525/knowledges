                # 将结果分页到热内存层以便立即使用
                self.memory.page_in(op["content"], k=3)

            elif op["type"] == "write":
                self.memory.write(
                    op["content"],
                    importance=0.8,  # 显式写入 = 重要
                    tier=MemoryTier.WARM,
                )

            elif op["type"] == "reflect":
                self._reflect()
