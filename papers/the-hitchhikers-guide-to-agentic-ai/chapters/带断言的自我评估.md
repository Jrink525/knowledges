        # 带断言的自我评估  
        assessment = self.assess(  
            context=context,  
            question=question,  
            answer=prediction.answer,  
        )  
        dspy.Assert(  
            assessment.faithful,  
            "Answer not faithful to context "  
            "(confidence: " + str(assessment.confidence) + ")"  
            "答案不忠实于上下文 (confidence: " + str(assessment.confidence) + ")"  
        )  
        return prediction
